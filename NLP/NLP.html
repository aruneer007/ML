def bag_of_words(text, W):
    bag = []
    token = nltk.word_tokenize(text)
    w_temp = [stemmer.stem(w.lower()) for w in token]
    
    for w in W:
        if w in w_temp:
            bag.append(1)
        else:
            bag.append(0)
    return bag
